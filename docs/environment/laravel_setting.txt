****************************************
Laravel(ver13)インストール
PHP8.5＋MySQL8.4
****************************************

# Laravelインストール
$ docker compose exec -w /var/www/html app create-project laravel/laravel --prefer-dist "13.*"

# npmインストール
$ docker compose exec -w /var/www/html app npm install

# npm実行
$ docker compose exec -w /var/www/html app npm run dev
# Ctrl+Cで終了

# WSLのIP確認
$ wsl -e hostname -I

# powershellからWindowsのhosts設定
$ powershell -NoProfile -ExecutionPolicy unrestricted -Command "start notepad C:\Windows\System32\drivers\etc\hosts -verb runas"

************************************
Laravel 初期設定
************************************

# config/app.php は.envのAPP_LOCALE=jaが優先される

# .env設定
APP_URL=https://base.local

APP_TIMEZONE=Asia/Tokyo
APP_LOCALE=ja
APP_FALLBACK_LOCALE=ja
APP_FAKER_LOCALE=ja_JP

LOG_CHANNEL=daily

DB_CONNECTION=mysql # mysql
DB_HOST=db # Dockerコンテナ名を指定
DB_PORT=3306 # Dockerコンテナポートを指定
DB_DATABASE=base
DB_USERNAME=root
DB_PASSWORD=passwd

REDIS_CLIENT=phpredis
REDIS_HOST=redis # Dockerコンテナ名を指定
REDIS_PORT=6379  # Dockerコンテナポートを指定
CACHE_STORE=redis
QUEUE_CONNECTION=redis

# 各機能でRedisを使用するように変更
CACHE_STORE=redis
SESSION_DRIVER=redis
QUEUE_CONNECTION=redis

MAIL_MAILER=smtp # smtp
MAIL_HOST=mailpit # Dockerコンテナ名を指定
MAIL_PORT=1025 # Dockerコンテナポートを指定

# composer.json設定
"require": {
    "php": "^8.5",
},

# migration実行
$ docker compose exec -w /var/www/html app php artisan migrate

# ide-helperのインストール
$ docker compose exec -w /var/www/html app composer require --dev barryvdh/laravel-ide-helper

# ide-helperファイルの生成
$ docker compose exec -w /var/www/html app php artisan ide-helper:generate

# 言語ファイルインストール
$ docker compose exec -w /var/www/html app composer require laravel-lang/common --dev
$ docker compose exec -w /var/www/html app php artisan lang:add ja

# Redisインストール
$ docker compose exec -w /var/www/html app composer require predis/predis

# シンボリックリンク作成
$ docker compose exec -w /var/www/html app php artisan storage:link

# 所有者変更
$ docker compose exec app chown www-data:www-data /var/www
$ docker compose exec app chown -R www-data:www-data /var/www/html

************************************
Laravel 疎通確認
************************************

# DB
$ docker compose exec app php artisan tinker
> DB::connection()->getPdo();

# mailpitに送信
$ docker compose exec app php artisan tinker
> Mail::raw('Mailpit Test', fn($m) => $m->to('test@example.com')->subject('Hello Mailpit'));

# Laravelのschedule:testを使う
$ docker compose logs -f app | grep "\[SCHEDULER\]"
$ docker compose exec app php artisan schedule:test

$ job
$ docker compose logs -f app | grep "\[WORKER\]"
$ docker compose exec app php artisan tinker --execute="Artisan::queue('about')"

# redis
$ docker compose exec app php artisan tinker
> \Illuminate\Support\Facades\Redis::ping();
